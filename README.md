## Install

```bash
npm install --save react-insta-chat
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'react-insta-chat'
import 'react-insta-chat/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
